1 Overview

Understanding the resource usage of a workflow is important for managing compute and storage resources.

There are tools in R to profile memory consumption. CPU, disk and storage activities are also important to measure. For linux systems, the collectl programs conveniently measure consumption in user-selectable time slices.

2 Demonstration

2.1 Data production and clustering by various algorithms

library(Rcollectl)
## 1/22 packages newly attached/loaded, see sessionInfo() for details.
library(BiocParallel)
## 1/2 packages newly attached/loaded, see sessionInfo() for details.
cl = cl_start()
cl_timestamp(cl,"make 5e6x100 data")
 mat = matrix(rnorm(5e6), ncol=100)
cl_timestamp(cl,"start 10k dist")
 dd = dist(mat[seq_len(1e4),])
cl_timestamp(cl,"start hclust")
 z = hclust(dd)
 library(bluster)
## 1/9 packages newly attached/loaded, see sessionInfo() for details.
cl_timestamp(cl,"start 10k clusterRows")
 ee = clusterRows(mat[seq_len(1e4),], KNNGraphParam(BPPARAM=MulticoreParam()))
cl_timestamp(cl,"start 20k dist")
 dd = dist(mat[seq_len(2e4),])
cl_timestamp(cl,"start hclust")
 z = hclust(dd)
 library(bluster)
## 0/0 packages newly attached/loaded, see sessionInfo() for details.
cl_timestamp(cl,"start 20k clusterRows")
 ee = clusterRows(mat[seq_len(2e4),], KNNGraphParam(BPPARAM=MulticoreParam()))
cl_timestamp(cl,"start 20k leiden")
 ee = clusterRows(mat[seq_len(2e4),], KNNGraphParam(BPPARAM=MulticoreParam(),cluster.fun="leiden"))
cl_stop(cl)
